def emoji
    response = {
         :id => @emoji.id,
         :char => @emoji.char_dec,
         :dec_val => @emoji.decimal,
         :hex_val => @emoji.hex,
         :name => @emoji.name,
         :code_point => "https://codepoints.net/U+#{@emoji.hex}",
         :html_escape => "&#x#{@emoji.hex}",
         :ruby => "\\u{#{@emoji.hex}}",
         :python => "\\U#{@emoji.hex}",
         :c => "\\U#{@emoji.hex}",
         :css => "\\0#{@emoji.hex}"
    }
    render :json => response.to_json
end

def search
    search_term = params[:q]
    query = "name LIKE '%#{search_term}%'"
    response = build_list Emoji.where(query)
    render :json => response.to_json
end
